import time

def medir_tempo(func):
    def wrapper(*args, **kwargs):
        inicio = time.time()
        resultado = func(*args, **kwargs)
        fim = time.time()
        print(f"{func.__name__} executou em {fim-inicio:.4f}s")
        return resultado
    return wrapper

@medir_tempo
def processar_dados(n):
    return sum(range(n))

resultado = processar_dados(1000000)
print(f"Resultado: {resultado}")

import json, os

class GerenciadorTarefas:
    def __init__(self, arquivo="tarefas.json"):
        self.arquivo = arquivo
        self.tarefas = self.carregar()

    def carregar(self):
        if os.path.exists(self.arquivo):
            with open(self.arquivo) as f:
                return json.load(f)
        return []

    def adicionar(self, titulo):
        self.tarefas.append({"titulo": titulo, "feita": False})
        print(f"Tarefa adicionada: {titulo}")

    def listar(self):
        for i, t in enumerate(self.tarefas, 1):
            status = "Feita" if t["feita"] else "Pendente"
            print(f"{i}. [{status}] {t['titulo']}")

g = GerenciadorTarefas()
g.adicionar("Estudar Python")
g.adicionar("Fazer exercícios")
g.listar()
